        // --- Function to create a single Pokémon card ---

        function createPokemonCard(pokemon) {

            const card = document.createElement('div');

            card.classList.add('pokemon-card');

            card.dataset.id = String(pokemon.id);

            // Store raw name and types string for easier filtering

            card.dataset.name = pokemon.name.toLowerCase();

            card.dataset.types = pokemon.types.join(',').toLowerCase(); // e.g., "grass,poison"


            // --- Header ---

            const headerInfo = document.createElement('div');

            headerInfo.classList.add('pokemon-header-info');

            const nameElement = document.createElement('h2');

            nameElement.classList.add('pokemon-name');

            nameElement.textContent = pokemon.name;

            headerInfo.appendChild(nameElement);

            const idElement = document.createElement('div');

            idElement.classList.add('pokemon-id');

            idElement.textContent = `#${String(pokemon.id).padStart(3, '0')}`;

            headerInfo.appendChild(idElement);

            card.appendChild(headerInfo);


            // --- Sprite ---

            const spriteContainer = document.createElement('div');

            spriteContainer.classList.add('pokemon-sprite-container');

            const sprite = document.createElement('img');

            sprite.classList.add('pokemon-sprite');

            const regularSrc = `${baseRegularSpriteUrl}${pokemon.sprite_name}.gif`;

            const shinySrc = `${baseShinySpriteUrl}${pokemon.sprite_name}.gif`;

            sprite.src = regularSrc; sprite.alt = `${pokemon.name}`;

            sprite.dataset.regularSrc = regularSrc; sprite.dataset.shinySrc = shinySrc; sprite.dataset.state = 'regular';

            sprite.loading = 'lazy';

            sprite.onerror = function() { /* Error handler */

                 this.onerror=null; this.src='data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7'; this.alt = `${pokemon.name} (Sprite not found)`; this.style.border = '1px dashed #ccc'; this.style.backgroundColor = '#f8f8f8'; const button = this.nextElementSibling; if(button && button.classList.contains('shiny-toggle')) { button.disabled = true; button.style.opacity = 0.5; button.style.cursor = 'not-allowed'; }

            };

            spriteContainer.appendChild(sprite);


             // Shiny Toggle Button

            const shinyButton = document.createElement('button');

            shinyButton.classList.add('shiny-toggle'); shinyButton.textContent = 'Shiny';

            shinyButton.addEventListener('click', (event) => { /* Shiny toggle logic */

                const button = event.currentTarget; const image = button.previousElementSibling; if (!image || !image.matches('.pokemon-sprite')) { console.error("Could not find sprite image for button", button); return; } if (image.dataset.state === 'regular') { image.src = image.dataset.shinySrc; image.dataset.state = 'shiny'; image.alt = `Shiny ${pokemon.name}`; button.textContent = 'Reg'; button.classList.add('active'); } else { image.src = image.dataset.regularSrc; image.dataset.state = 'regular'; image.alt = `${pokemon.name}`; button.textContent = 'Shiny'; button.classList.remove('active'); }

            });

            spriteContainer.appendChild(shinyButton);

            card.appendChild(spriteContainer);


            // --- Types ---

            const typesContainer = document.createElement('div');

            typesContainer.classList.add('pokemon-types');

            pokemon.types.forEach(type => {

                const typeSpan = document.createElement('span');

                typeSpan.textContent = type;

                const typeClass = `type-${type.toLowerCase()}`;

                typeSpan.classList.add(typeClass);

                // Add data attribute for easy type extraction on click

                typeSpan.dataset.type = type.toLowerCase();

                typesContainer.appendChild(typeSpan);

            });

            card.appendChild(typesContainer);


             // --- Stats ---

            const statsList = document.createElement('ul');

            statsList.classList.add('pokemon-stats');

            const statsMap = { 'HP': pokemon.stats.hp, 'Atk': pokemon.stats.attack, 'Def': pokemon.stats.defense, 'Sp.A': pokemon.stats.sp_attack, 'Sp.D': pokemon.stats.sp_defense, 'Spe': pokemon.stats.speed };

            for (const [statName, statValue] of Object.entries(statsMap)) {

                const statItem = document.createElement('li'); const statNameSpan = document.createElement('span'); const statValueSpan = document.createElement('span'); statNameSpan.textContent = statName; statValueSpan.textContent = statValue; statItem.appendChild(statNameSpan); statItem.appendChild(statValueSpan); statsList.appendChild(statItem);

            }

            card.appendChild(statsList);


            return card;

        }


        // --- Function to display Pokémon cards ---

        function displayPokemon(pokemonList) {

            if (!pokedexGrid) return;

            pokedexGrid.innerHTML = ''; // Clear grid

            allPokemonCards = []; // Reset card references

            pokemonList.forEach(pokemon => {

                try {

                    const card = createPokemonCard(pokemon);

                    pokedexGrid.appendChild(card);

                    allPokemonCards.push(card); // Store reference

                } catch (error) {

                    console.error(`Error creating card for Pokémon ID ${pokemon.id} (${pokemon.name}):`, error);

                     const errorCard = document.createElement('div'); errorCard.classList.add('pokemon-card'); errorCard.style.backgroundColor = '#ffdddd'; errorCard.textContent = `Error loading ${pokemon.name} (ID: ${pokemon.id}). Check console.`; pokedexGrid.appendChild(errorCard);

                }

            });

            applyFilters(); // Apply any active filters after display

        }



        // --- Function to apply all active filters (Search + Type) ---

        function applyFilters() {

            const searchTerm = searchBar.value.toLowerCase().trim();

            let matchFound = false;

            let isRandomMode = false; // Flag to check if random mode is active


            // Check if the grid has only one child and it's visible (potential random mode)

            if (pokedexGrid.childElementCount === pokemonData.length) { // Only check if seems like full grid is present

                 const visibleCards = allPokemonCards.filter(card => !card.classList.contains('hidden'));

                 if (visibleCards.length === 1) {

                    // It might be random mode, but let's re-apply filters normally

                    // unless specifically triggered by random button. We'll handle

                    // random display separately.

                 }

            }



            // Apply search and type filters

            allPokemonCards.forEach(card => {

                const name = card.dataset.name;

                const id = card.dataset.id;

                const types = card.dataset.types; // e.g., "grass,poison"


                const searchMatch = name.includes(searchTerm) || id.includes(searchTerm);

                const typeMatch = !activeTypeFilter || types.includes(activeTypeFilter);


                if (searchMatch && typeMatch) {

                    card.classList.remove('hidden'); // Use class for visibility

                    matchFound = true;

                } else {

                    card.classList.add('hidden');

                }

            });


            // Update the 'active-filter' class on type spans

            document.querySelectorAll('.pokemon-types span').forEach(span => {

                if (activeTypeFilter && span.dataset.type === activeTypeFilter) {

                    span.classList.add('active-filter');

                } else {

                    span.classList.remove('active-filter');

                }

            });



            noResultsMsg.style.display = matchFound ? 'none' : 'block';

        }


        // --- Event Handlers ---


        // Search Bar Input

        searchBar.addEventListener('input', applyFilters);


        // Show All / Clear Filters Button

        showAllBtn.addEventListener('click', () => {

            searchBar.value = '';        // Clear search

            activeTypeFilter = null;     // Clear type filter

            applyFilters();              // Re-apply (which shows all)

            // Ensure all cards are potentially visible before filtering

            allPokemonCards.forEach(card => card.classList.remove('hidden'));

            applyFilters(); // Re-apply filters to show all

        });


        // Random Pokémon Button

        randomBtn.addEventListener('click', () => {

            searchBar.value = '';        // Clear search

            activeTypeFilter = null;     // Clear type filter

            noResultsMsg.style.display = 'none'; // Hide no results message


            const randomIndex = Math.floor(Math.random() * pokemonData.length);

            const randomPokemonId = String(pokemonData[randomIndex].id);


            allPokemonCards.forEach(card => {

                if (card.dataset.id === randomPokemonId) {

                    card.classList.remove('hidden');

                    card.scrollIntoView({ behavior: 'smooth', block: 'center' }); // Scroll to the card

                } else {

                    card.classList.add('hidden');

                }

            });

            // Clear active type filter visuals

             document.querySelectorAll('.pokemon-types span.active-filter').forEach(span => {

                span.classList.remove('active-filter');

             });

        });


        // Type Filter Click (Event Delegation)

        pokedexGrid.addEventListener('click', (event) => {

            const clickedTypeSpan = event.target.closest('.pokemon-types span');

            if (clickedTypeSpan) {

                const type = clickedTypeSpan.dataset.type;

                if (activeTypeFilter === type) {

                    // Optional: Clicking the same type again could clear the filter

                    // activeTypeFilter = null;

                } else {

                    activeTypeFilter = type;

                }

                applyFilters();

            }

        });


        // --- Initial Display ---

        if (!pokedexGrid) {

            console.error("FATAL ERROR: Could not find element with ID 'pokedex-grid'.");

        } else {

            console.log("Initializing Pokédex...");

            displayPokemon(pokemonData); // Display all Pokémon initially

        }


    </script>
